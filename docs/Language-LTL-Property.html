nonnegative :: Formula -> Bool Source

is the first operator not a negation ?

>>> nonnegative $ Binary U (AP "p") (Unary G (AP "q"))
True

>>> nonnegative $ Unary Neg $ Binary U (AP "p") (Unary G (AP "q"))
False
